// Debug function for navigation window.debugNavigation = function() { const navInfo = { totalButtons: document.querySelectorAll('.nav-item').length, yellowButtons: document.querySelectorAll('.nav-item:not(.campus-3d)').length, greenButtons: document.querySelectorAll('.nav-item.campus-3d').length, activeButton: document.querySelector('.nav-item.active')?.textContent, responsiveBreakpoints: { desktop: window.innerWidth > 992, tablet: window.innerWidth <= 992 && window.innerWidth > 768, mobile: window.innerWidth <= 768 } }; console.log('🔍 Navigation Debug:', navInfo); return navInfo; };
